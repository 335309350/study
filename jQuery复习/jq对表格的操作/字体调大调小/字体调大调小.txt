通过用css("font-size")获取当前元素的字体大小和单位，然后分割用parseInt(txt_size,10)转化为数字大小，获取单位用txt_size.slice(-2)截取后两位字符串，通过点击调大，调小按钮来触发事件，给元素设置字体大小用css();

核心代码：
//调大字体
		$("#content_font span").click(function(){
			var txt_size=$(".msg_txt").css("font-size");
			//获取文本域中的默认字体大小和单位
			//alert(txt_size);
			var txt_font=parseInt(txt_size,10);
			//alert(txt_font);
			//得到字体的大小数字
			//获取单位，截取后面两个字符串
			var txt_unit=txt_size.slice(-2);
			//alert(txt_unit);
			//点击调大和调小获取对象的class
			var cName=$(this).attr("class");
			//alert(cName);
			//根据当前对象的class执行不同的事件
			if(cName=="bigger_font"){
				if(txt_font<=22)
					txt_font+=2;
					//最大字体为22px
			
			}else{
				if(txt_font>16)
				txt_font-=2;
				//最小字体是16px
			}
			//拼接设置字体
			$(".msg_txt").css("font-size",txt_font+txt_unit);
			//字体数字+单位
		});
