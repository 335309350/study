	<script type="text/javascript">
		//jq延迟动画
		$(function(){
			$("#panel").click(function(){
				//alert("ok");
				$(this).animate({height:"200px",left:"20px",opacity:"1"},1000).delay(2000).animate({top:"50px",width:"400px"},1000).delay(2000).fadeOut(3000);
				//向右边移动20px，设置不透明度是1后延迟2s后向下移动50px,宽度变为400px,延迟3s后消失
			
			});
			
			
		});
	
	</script>
