点击弹出一个层
在外面套一个盒子，作为拖动。点击某个按钮弹出这个盒子，利用css的display来实现
<!DOCTYPE HTML>
<html lang="en-US">
<head>
	<meta charset="UTF-8">
	<title>jq弹出一个层</title>
	<script type="text/javascript" src="./js/jquery-1.8.3.min.js"></script>
	<script type="text/javascript" src="./js/jquery.easydrag.handler.beta2.js"></script>
	<script type="text/javascript">
		$(function(){
			$(".admin dd a").click(function(){
				//点击弹出一个层
				//$(".addRole").css("display","block");
				//具有动画效果
				$("#addRole").fadeIn();
			});
			$(".one").click(function(){
				//点击弹出一个层
				//$(".addRole").css("display","block");
				//具有动画效果
				$("#addRole").fadeIn();
			});
			//点击关闭层
			$("#addRole span a").click(function(){
				//$(".addRole").css("display","none");
				$("#addRole").hide();
			});
			//鼠标放上去移动框
			$(".addRole_box").easydrag();
		});
	</script>
	<style type="text/css">
		*{padding:0;margin:0;}
		body{margin:0;padding:font-size:16px;}
		#addRole{width:300px;height:300px;border:1px solid #eee;margin-left:120px;display:none;box-shadow:1px 2px 3px green;}
		#addRole h1{line-height:30px;height:30px;font-size:14px;background:#abcdef;padding-left:12px;color:#000;}
		#addRole span{float:right;padding-right:12px;cursor:pointer;margin-top:-22px;}
		#addRole span a{text-decoration:none;color:#fff;}
		#addRole p{padding:12px;text-indent:2em;font-size:14px;}
		
	</style>
</head>

		<dt>管理员操作</dt>
		<dd><a href="#">添加管理员</a></dd>
	</dl>

	<div class="addRole_box">
		<div id="addRole">
			<h1>这是标题</h1>
			<span><a href="#">关闭</a></span>
			<p>这是内容</p>
		</div>
	</div>
</body>
